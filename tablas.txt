CREATE TABLE IF NOT EXISTS public.reservas_cliente
(
    id_cliente integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    nombre character varying(40) COLLATE pg_catalog."default" NOT NULL,
    tel character varying(13) COLLATE pg_catalog."default" NOT NULL,
    correo character varying(30) COLLATE pg_catalog."default" NOT NULL,
    departamento character varying(30) COLLATE pg_catalog."default" NOT NULL,
    ciudad character varying(30) COLLATE pg_catalog."default" NOT NULL,
    edad character varying(10) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT reservas_cliente_pkey PRIMARY KEY (id_cliente)
)

CREATE TABLE IF NOT EXISTS public.reservas_solicitud
(
    id_solicitud integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    fecha date NOT NULL,
    no_personas character varying(10) COLLATE pg_catalog."default" NOT NULL,
    motivo character varying(40) COLLATE pg_catalog."default" NOT NULL,
    observaciones character varying(200) COLLATE pg_catalog."default" NOT NULL,
    estado character varying(15) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT reservas_solicitud_pkey PRIMARY KEY (id_solicitud)
)
